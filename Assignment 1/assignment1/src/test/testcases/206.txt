
      Class Spider_verse {
         Val Suits: String;
         Var name: String;
         Val Num: Int;
         
         getSuits() {
            Return Self.Suits;
         }
         
         getName() {
            Return Self.name;
         }
         
         getNum() {
            Return Self.Num;
         }
         
         Constructor(Suits: String; name: String; Num: Int) {
            Self.Suits = Suits;
            Self.name = name;
            Self.Num = Num;
         }
         Destructor() {
         
         }
      }
         
      Class Program {
         main() {
            Var Peter_1: Spider_verse = New Spider_verse("Red", "Peter Parker", 1);
            Val Peter_1_Suits: String = Peter_1.getSuits();
            Var Peter_1_name: String = Peter_1.getName();
         }
      }
