
         Class testAlgorithms {
            binarySearch(arr: Array[Int, 10]; l: Int; r: Int; x: Int) 
            { 
               If (r >= l) { 
                  Var mid: Int = l + (r - l) / 2;  
                  If (arr[mid] == x) {
                     Return mid;
                  }
                  If (arr[mid] > x) {
                     Return Self.binarySearch(arr, l, mid - 1, x);
                  }
                  Return Self.binarySearch(arr, mid + 1, r, x); 
               } 
               Return -1; 
            } 
         }
